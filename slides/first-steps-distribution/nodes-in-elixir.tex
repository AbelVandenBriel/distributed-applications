\begin{frame}
    \frametitle{Give your node a name}
    Two options:
    \begin{itemize}
        \item short name: \texttt{name@hostname}
        \item long name: fully qualified name (with ip-address for example)
    \end{itemize}
    
    \vfill

    Check this with \texttt{Node.self/0}
\end{frame}
